Python 代码示例

def fibonacci(n):
    """计算斐波那契数列的第n项"""
    if n <= 1:
        return n

    a, b = 0, 1
    for i in range(2, n + 1):
        a, b = b, a + b

    return b

# 测试函数
for i in range(10):
    print(f"F({i}) = {fibonacci(i)}")
